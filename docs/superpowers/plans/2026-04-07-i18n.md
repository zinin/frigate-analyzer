# I18n Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add multi-language support (ru/en) to the Telegram bot with automatic language detection, manual override, and per-user localized notifications.

**Architecture:** Spring `MessageSource` with `.properties` files for translations. New `language_code` column in `telegram_users`. `MessageResolver` service wraps `MessageSource` and is injected into all handlers/services. `UserZoneInfo` extended with `language` field for per-user notification localization.

**Tech Stack:** Kotlin, Spring Boot, R2DBC, Liquibase, ktgbotapi

sortedHandlers.forEach { handler ->
    onCommand(handler.command, requireOnlyCommandInMessage = false) { message ->
        val user: TelegramUserDto? =
            if (handler.requiredRole != null) {
                val username = authorizationFilter.extractUsername(message)
                if (username == null) {
                    val telegramLang = (message as? PrivateContentMessage<*>)?.user?.languageCode
                    val lang = StartCommandHandler.detectLanguage(telegramLang)
                    reply(message, msg.get("common.error.unauthorized", lang))
                    return@onCommand
                }

                val resolvedUser = userService.findActiveByUsername(username)
                val resolvedRole = when {
                    username == properties.owner -> UserRole.OWNER
                    resolvedUser != null -> UserRole.USER
                    else -> null
                }

                if (resolvedRole == null) {
                    val telegramLang = (message as? PrivateContentMessage<*>)?.user?.languageCode
                    val lang = StartCommandHandler.detectLanguage(telegramLang)
                    reply(message, msg.get("common.error.unauthorized", lang))
                    return@onCommand
                }

                if (handler.requiredRole == UserRole.OWNER && resolvedRole != UserRole.OWNER) {
                    val lang = resolvedUser?.languageCode ?: "ru"
                    reply(message, msg.get("common.error.owner.only", lang))
                    return@onCommand
                }

                resolvedUser
            } else {
                null
            }

        try {
            with(handler) { handle(message, user) }
        } catch (e: CancellationException) {
            throw e
        } catch (e: Exception) {
            logger.error(e) { "Error handling command /${handler.command}" }
            val lang = user?.languageCode
                ?: StartCommandHandler.detectLanguage((message as? PrivateContentMessage<*>)?.user?.languageCode)
            reply(message, msg.get("common.error.generic", lang))
        }
    }

    @Transactional(readOnly = true)
    override suspend fun getUserLanguage(chatId: Long): String {
        val user = repository.findByChatId(chatId)
        return user?.languageCode ?: "ru"
    }

    @Transactional
    override suspend fun updateLanguage(chatId: Long, languageCode: String): Boolean {
        val updated = repository.updateLanguageCode(chatId, languageCode)
        if (updated == 0L) {
            logger.warn { "updateLanguage: no rows updated for chatId=$chatId" }
            return false
        }
        logger.info { "Updated language for chatId=$chatId to $languageCode" }
        return true
    }

- [ ] **Step 1: Create `MessageResolver`**

```kotlin
package ru.zinin.frigate.analyzer.telegram.i18n

import io.github.oshai.kotlinlogging.KotlinLogging
import org.springframework.context.MessageSource
import org.springframework.context.NoSuchMessageException
import org.springframework.stereotype.Component
import java.util.Locale

private val logger = KotlinLogging.logger {}

@Component
class MessageResolver(private val messageSource: MessageSource) {

    fun get(key: String, language: String, vararg args: Any): String =
        get(key, Locale.forLanguageTag(language), *args)

    fun get(key: String, locale: Locale, vararg args: Any): String =
        try {
            messageSource.getMessage(key, args, locale)
        } catch (e: NoSuchMessageException) {
            logger.warn { "Missing translation key='$key' for locale='$locale'" }
            try {
                messageSource.getMessage(key, args, DEFAULT_LOCALE)
            } catch (e: NoSuchMessageException) {
                key
            }
        }

    companion object {
        private val DEFAULT_LOCALE = Locale.forLanguageTag("ru")
    }
}
```

- [ ] **Step 2: Create `I18nConfiguration`**

```kotlin
package ru.zinin.frigate.analyzer.telegram.i18n

import org.springframework.context.MessageSource
import org.springframework.context.annotation.Bean
import org.springframework.context.annotation.Configuration
import org.springframework.context.support.ReloadableResourceBundleMessageSource

@Configuration
class I18nConfiguration {

    @Bean
    fun messageSource(): MessageSource =
        ReloadableResourceBundleMessageSource().apply {
            setBasename("classpath:messages")
            setDefaultEncoding("UTF-8")
            setFallbackToSystemLocale(false)
            setDefaultLocale(java.util.Locale.forLanguageTag("ru"))
        }
}
```

                                }
                            },

                    waitDataCallbackQuery()

                                (it as? MessageDataCallbackQuery)?.message?.let { m ->
                                    m.messageId == sentMessage.messageId && m.chat.id == chatId
                                } == true

                    return@withTimeoutOrNull

                sendTextMessage(chatId, msg.get("command.language.set", newLang, langName))
            }

            sendTextMessage(chatId, msg.get("command.language.timeout", lang))
